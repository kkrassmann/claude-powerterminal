```typescript
import { Injectable } from '@angular/core';
import { ProjectAuditResult } from '../../../../shared/audit-types';

declare const window: any;

@Injectable({ providedIn: 'root' })
export class AuditService {

  async loadProjects(): Promise<string[]> {
    if (window.electronAPI) {
      return window.electronAPI.invoke('audit:projects');
    }
    const resp = await fetch(
      `http://${window.location.hostname}:9801/api/audit/projects`
    );
    if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
    return resp.json();
  }

  async runAudit(projectPath: string): Promise<ProjectAuditResult> {
    if (window.electronAPI) {
      return window.electronAPI.invoke('audit:run', projectPath);
    }
    const resp = await fetch(
      `http://${window.location.hostname}:9801/api/audit/run?path=${encodeURIComponent(projectPath)}`
    );
    if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
    return resp.json();
  }

  formatProjectName(projectPath: string): string {
    // Extract last two path segments for display: "Dev/my-project"
    const parts = projectPath.replace(/\\/g, '/').split('/').filter(Boolean);
    return parts.length >= 2 ? parts.slice(-2).join('/') : projectPath;
  }
}
```

   ```typescript
       this.loadAuditProjects();
     }
   }
   ```

   ```typescript
   async startAudit(): Promise<void> {
     if (!this.selectedProject || this.auditLoading) return;
     this.auditLoading = true;
     this.auditError = null;
     this.expandedFiles.clear();
     try {
       this.auditResult = await this.auditService.runAudit(this.selectedProject);
     } catch (err) {
       this.auditError = String(err);
     } finally {
       this.auditLoading = false;
     }
   }
   ```

   ```typescript
   toggleFile(filePath: string): void {
     if (this.expandedFiles.has(filePath)) {
       this.expandedFiles.delete(filePath);
     } else {
       this.expandedFiles.add(filePath);
     }
   }
   ```
